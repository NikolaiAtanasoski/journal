# Keypirinha #

## Keypirinha.ini ##
```
[app]
hotkey_run = Alt+Space
ignored_packages = <ALL>, -Internal, -FilesCatalog, -Apps, -Kill
```

[profile/PortableApps]
activate = yes
max_depth = 2
paths = 
     C:\Users\User\Desktop\**
     C:\install\**
     D:\tools\**
     D:\dokumente\**
     D:\Downloads\**
     D:\scripts\**
     D:\startup\**
     D:\notizen\**
filters = - ext: .jar
